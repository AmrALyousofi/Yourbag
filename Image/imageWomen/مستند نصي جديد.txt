// تفعيل الموقع عند التحميل
document.addEventListener('DOMContentLoaded', function() {
    // إضافة سنة الحالية في التذييل
    document.getElementById('current-year').textContent = new Date().getFullYear();
    
    // تفعيل القائمة على الهاتف
    const mobileMenuBtn = document.querySelector('.mobile-menu-btn');
    const mainNav = document.querySelector('.main-nav');
    
    if (mobileMenuBtn) {
        mobileMenuBtn.addEventListener('click', function() {
            mainNav.classList.toggle('active');
        });
    }
    
    // تفعيل التنقل السلس
    document.querySelectorAll('a[href^="#"]').forEach(anchor => {
        anchor.addEventListener('click', function(e) {
            const href = this.getAttribute('href');
            
            // تجاهل الروابط التي لا تشير إلى أقسام
            if (href === '#') return;
            
            // إذا كان الرابط لقسم في نفس الصفحة
            if (href.startsWith('#')) {
                e.preventDefault();
                
                const targetId = href.substring(1);
                const targetElement = document.getElementById(targetId);
                
                if (targetElement) {
                    // إغلاق القائمة على الهاتف
                    mainNav.classList.remove('active');
                    
                    // إضافة active للرابط المختار
                    document.querySelectorAll('.nav-list a').forEach(a => {
                        a.classList.remove('active');
                    });
                    this.classList.add('active');
                    
                    // التمرير السلس للقسم المطلوب
                    window.scrollTo({
                        top: targetElement.offsetTop - 80,
                        behavior: 'smooth'
                    });
                }
            }
        });
    });
    
    // تفعيل أزرار عرض التفاصيل
    document.querySelectorAll('.product-card .btn').forEach(button => {
        button.addEventListener('click', function() {
            const productCard = this.closest('.product-card');
            const productName = productCard.querySelector('.product-name').textContent;
            const productPrice = productCard.querySelector('.price').textContent;
            
            // في الموقع الحقيقي، يمكن توجيه المستخدم لصفحة تفاصيل المنتج
            // هنا سنعرض رسالة توضيحية
            alert(`تفاصيل المنتج:\n\nالاسم: ${productName}\nالسعر: ${productPrice}\n\nسيتم تطوير صفحة تفاصيل المنتج في المرحلة القادمة.`);
        });
    });
    
    // تفعيل التنقل النشط حسب التمرير
    function updateActiveNav() {
        const sections = document.querySelectorAll('section[id]');
        const navLinks = document.querySelectorAll('.nav-list a');
        
        let current = '';
        
        sections.forEach(section => {
            const sectionTop = section.offsetTop - 100;
            const sectionHeight = section.clientHeight;
            const sectionId = section.getAttribute('id');
            
            if (scrollY >= sectionTop && scrollY < sectionTop + sectionHeight) {
                current = sectionId;
            }
        });
        
        navLinks.forEach(link => {
            link.classList.remove('active');
            if (link.getAttribute('href') === `#${current}`) {
                link.classList.add('active');
            }
        });
    }
    
    window.addEventListener('scroll', updateActiveNav);
    
    // تحميل الصور بكسل بطيء (Lazy Loading)
    const images = document.querySelectorAll('img[loading="lazy"]');
    
    if ('IntersectionObserver' in window) {
        const imageObserver = new IntersectionObserver((entries, observer) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const img = entry.target;
                    img.src = img.src;
                    imageObserver.unobserve(img);
                }
            });
        });
        
        images.forEach(img => imageObserver.observe(img));
    }
});
